find_col <- function(df, patterns) {

  hits <- names(df)[str_detect(
    tolower(names(df)),
    paste(tolower(patterns), collapse = "|")
  )]

  if (length(hits) == 0) {
    stop(
      "Could not find column matching: ",
      paste(patterns, collapse = ", ")
    )
  }

  hits[1]
}
# Convert BED-like peak table to GRanges

make_gr <- function(df) {

  chr_col   <- find_col(df, c("^chr$", "chrom", "chromosome"))
  start_col <- find_col(df, c("^start$"))
  end_col   <- find_col(df, c("^end$"))

  GRanges(
    seqnames = df[[chr_col]],
    ranges = IRanges(
      start = as.integer(df[[start_col]]) + 1,
      end   = as.integer(df[[end_col]])
    )
  )
}

mean_meth_in_region <- function(meth_gr, regions_gr) {
  hits <- findOverlaps(regions_gr, meth_gr)
  result <- rep(NA_real_, length(regions_gr))
  
  if (length(hits) > 0) {
    tmp <- data.frame(
      region_id = queryHits(hits),
      meth = mcols(meth_gr)$score[subjectHits(hits)]
    )
    
    region_means <- tapply(tmp$meth, tmp$region_id, mean, na.rm = TRUE)
    
    result[as.integer(names(region_means))] <- as.numeric(region_means)
  }
  
  result
}
